    <div id="clicker-container">
        <h1>Advanced Clicker</h1>
        <p>Clicks: <span id="click-count">0</span></p>
        <button id="click-button">Click Me!</button>
        <p id="message"></p>
    </div>

    <script>
        document.addEventListener('DOMContentLoaded', () => {
            const clickButton = document.getElementById('click-button');
            const clickCountSpan = document.getElementById('click-count');
            const messageParagraph = document.getElementById('message');
            let clickCount = 0;

            const messages = {
                1: "First click! Keep going!",
                10: "You've reached 10 clicks!",
                25: "Wow, 25 clicks already!",
                50: "Half a hundred! Impressive!",
                100: "Congratulations! You hit 100 clicks!"
            };

            clickButton.addEventListener('click', () => {
                clickCount++;
                clickCountSpan.textContent = clickCount;

                if (messages[clickCount]) {
                    messageParagraph.textContent = messages[clickCount];
                } else if (clickCount > 100 && clickCount % 50 === 0) {
                     messageParagraph.textContent = `You're on a roll! Reached ${clickCount} clicks!`;
                } else if (clickCount > 100) {
                     messageParagraph.textContent = "";
                }
            });
        });
    </script>
